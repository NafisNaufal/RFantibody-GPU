\begin{frame}{Three Specific Bottlenecks Identified}
\vspace{0.3cm}
\begin{columns}[T]
\begin{column}{0.32\textwidth}
  \begin{block}{\textcolor{CPURed}{1.} \ SO(3) Rotation Sampling}
    \small
    Called \textbf{49$\times$} per design via \texttt{scipy}. Pulled all rotation matrices to CPU, converted to axis-angle, sampled noise, then pushed back to GPU.
  \end{block}
\end{column}
\begin{column}{0.32\textwidth}
  \begin{block}{\textcolor{CPURed}{2.} \ Kabsch Superposition}
    \small
    Motif alignment used \texttt{numpy} SVD every timestep. Required \textbf{three full tensor copies} to CPU and back per step.
  \end{block}
\end{column}
\begin{column}{0.32\textwidth}
  \begin{block}{\textcolor{CPURed}{3.} \ Per-step Logging Sync}
    \small
    Each of the 50 log lines called \texttt{.tolist()} on a GPU tensor — forcing the GPU to \textbf{flush its pipeline} before proceeding.
  \end{block}
\end{column}
\end{columns}
\vspace{0.5cm}
\begin{center}
\small\textcolor{MidGray}{All three happen inside the 50-timestep diffusion loop — their cost is paid 49--50 times per design.}
\end{center}
\end{frame}

% 4. SOLUTIONS
% ══════════════════════════════════════════════════════════════════════════════
\sectionslide{4. \ Solutions Implemented}

\begin{frame}{Four Targeted Changes — Model and Outputs Unchanged}
\vspace{0.3cm}
\begin{columns}[T]
\begin{column}{0.24\textwidth}
  \begin{block}{\textcolor{Green}{1.} \ SO(3) on GPU}
    \small
    Replaced \texttt{scipy} axis-angle conversion and \texttt{numpy} interpolation with PyTorch native ops. Noise stays on device.
  \end{block}
\end{column}
\begin{column}{0.24\textwidth}
  \begin{block}{\textcolor{Green}{2.} \ Kabsch on GPU}
    \small
    Replaced \texttt{numpy} SVD with \texttt{torch.linalg.svd}. All three CPU tensor copies eliminated.
  \end{block}
\end{column}
\begin{column}{0.24\textwidth}
  \begin{block}{\textcolor{Green}{3.} \ Device Fixes}
    \small
    Corrected tensor device mismatches throughout the denoising loop — removed silent CPU fallbacks.
  \end{block}
\end{column}
\begin{column}{0.24\textwidth}
  \begin{block}{\textcolor{Green}{4.} \ TF32 + Log Fix}
    \small
    Enabled TF32 matmul on tensor cores. Removed \texttt{.tolist()} GPU flush from per-step log.
  \end{block}
\end{column}
\end{columns}
\vspace{0.5cm}
\begin{center}
  \small\textcolor{MidGray}{No changes to model weights, architecture, or numerical outputs.}
\end{center}
\end{frame}
