\begin{frame}{Three Specific Bottlenecks Were Identified}
\vspace{0.3cm}
\begin{columns}[t]
  \begin{column}{0.32\textwidth}
    \begin{block}{1.\ \ SO(3) Rotation Sampling}
      \small
      Called 49$\times$ per design via \texttt{scipy}.\\[4pt]
      Pulled all rotation matrices to CPU, converted to axis-angle, sampled noise on CPU, then pushed back to GPU.
    \end{block}
  \end{column}
  \begin{column}{0.32\textwidth}
    \begin{block}{2.\ \ Kabsch Superposition}
      \small
      Motif alignment used \texttt{numpy} SVD every timestep.\\[4pt]
      Three full tensor copies CPU $\leftrightarrow$ GPU per step.
    \end{block}
  \end{column}
  \begin{column}{0.32\textwidth}
    \begin{block}{3.\ \ Per-step Logging Sync}
      \small
      Each of 50 log lines called \texttt{.tolist()} on a GPU tensor.\\[4pt]
      Forces the GPU to flush its pipeline before continuing.
    \end{block}
  \end{column}
\end{columns}
\vspace{0.5cm}
\begin{center}
  \small\color{MidGray}
  Each bottleneck repeated \textbf{49--50 times per design} — $\sim$1,000 unnecessary CPU/GPU round-trips for a 1,000-design campaign.
\end{center}
\end{frame}

\begin{frame}{Four Targeted Changes — Model and Outputs Unchanged}
\vspace{0.2cm}
\begin{columns}[t]
  \begin{column}{0.24\textwidth}
    \begin{block}{SO(3) $\rightarrow$ GPU}
      \scriptsize\color{MidGray}\texttt{diffusion.py}\\[4pt]
      \small\color{NavyBlue}
      Replaced \texttt{scipy} axis-angle conversion and \texttt{numpy} interpolation with native PyTorch ops. Noise sampled on-device.
    \end{block}
  \end{column}
  \begin{column}{0.24\textwidth}
    \begin{block}{Kabsch $\rightarrow$ GPU}
      \scriptsize\color{MidGray}\texttt{utils.py}\\[4pt]
      \small\color{NavyBlue}
      Replaced \texttt{numpy} SVD and all \texttt{.cpu().numpy()} calls with \texttt{torch.linalg.svd}. Result stays on GPU.
    \end{block}
  \end{column}
  \begin{column}{0.24\textwidth}
    \begin{block}{Device Fixes}
      \scriptsize\color{MidGray}\texttt{utils.py}\\[4pt]
      \small\color{NavyBlue}
      Corrected tensor device mismatches silently causing CPU fallbacks throughout the denoising loop.
    \end{block}
  \end{column}
  \begin{column}{0.24\textwidth}
    \begin{block}{TF32 + Log Fix}
      \scriptsize\color{MidGray}\texttt{inference.py}\\[4pt]
      \small\color{NavyBlue}
      Enabled TF32 matmul on tensor cores. Removed \texttt{.tolist()} GPU pipeline flush from per-step logging.
    \end{block}
  \end{column}
\end{columns}
\vspace{0.5cm}
\begin{center}
  \small\color{MidGray}
  No changes to model weights, architecture, or numerical outputs.
\end{center}
\end{frame}

\begin{frame}{Summary}
\vspace{0.2cm}
\begin{columns}[t]
  \begin{column}{0.24\textwidth}
    \begin{block}{Root Cause}
      \small
      CPU--GPU round-trips every diffusion timestep starved the GPU pipeline — worsened by the EPYC 7742's low single-core clock.
    \end{block}
  \end{column}

    \begin{block}{What We Did}
      \small
      Ported SO(3) sampling and Kabsch alignment to native PyTorch. Fixed device inconsistencies and removed GPU sync from logging.
    \end{block}
  \end{column}

    \begin{block}{Result}
      \small
      \textbf{14\% faster} on RTX~3060 (10.5 $\to$ 9.0 min). GPU at 100\% utilisation — single-design ceiling reached.
    \end{block}
  \end{column}
